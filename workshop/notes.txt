=== 2_dockerfile_with_python ===
--------------------------------
FROM ubuntu
MAINTAINER adhese
RUN apt-get update && DEBIAN_FRONTEND=noninteractive apt-get -y install python
ADD slides slides
WORKDIR /slides
CMD python -m SimpleHTTPServer 8765
EXPOSE 8765
--------------------------------

docker build -t demo/slides .
docker run -d demo/slides
### show website (oops!)

docker ps 
### show ip (not mapped)

docker run -d -p 8765 demo/slides
docker ps 
### show ip (mapped!)
### show website (hurray!)



=== 3_volume ===
--- server -----
FROM ubuntu
MAINTAINER adhese
RUN apt-get update && DEBIAN_FRONTEND=noninteractive apt-get -y install python
WORKDIR /slides
CMD python -m SimpleHTTPServer 8765
EXPOSE 8765
----------------

--- website ----
FROM ubuntu
MAINTAINER adhese
VOLUME /slides
ADD slides slides
CMD echo "Volume only"
---------------- 

cd server
docker build -t demo/server .
cd ../website
docker build -t demo/website .
docker images
docker run demo/website                   ###notice without "-d" 
docker run -d -p 8765 demo/server 
### show server (oops!) - did not bind volume to server

docker run --name website demo/website 
docker run --volumes-from website -d -p 8765 demo/server 

### show works!
### we can start another application using the same server config

cd ../empty_website
docker build -t demo/empty_website .
docker run --name empty_website demo/empty_website
docker run --volumes-from empty_website -d -p 8765 demo/server 

### show empty website works
