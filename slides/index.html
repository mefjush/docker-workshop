### Volumes 
# https://docs.docker.com/userguide/dockervolumes/
docker run -d -P --name portal --link rates1:rates -v /portal/static workshop/portal
docker run --volumes-from portal ubuntu ls /portal/static
           
# what if we want to 're-deploy' portal?
docker stop portal
docker run --volumes-from portal ubuntu ls /portal/static
docker rm portal
docker run --volumes-from portal ubuntu ls /portal/static

### More 'persistent' volume 
docker create -v /portal/static --name portal_data ubuntu /bin/true
docker run -d -P --name portal --link rates1:rates --volumes-from portal_data workshop/portal
           

### Data migration
mkdir bak
docker run --volumes-from portal_data -v $(pwd)/bak:/backup ubuntu cp -r /portal/static /backup

Image best practices

  • stateless (need state -> volume)
  • minimalistic (no unnecessary stuff)
  • no SSHD "for debugging" (contraversive)
  • one process per container (contraversive)
